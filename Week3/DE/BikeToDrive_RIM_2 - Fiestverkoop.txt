--BikeToDrive_2_Fietsverkoop;

CREATE TABLE Fabrikant (
	fabrikantnr INT,
	naam VARCHAR(45),
	adres VARCHAR(100),
	plaats VARCHAR(45),

	PRIMARY KEY(fabrikantnr)
);

CREATE TABLE Filiaal (
	filiaalnr INT,
	naam VARCHAR(45),
	adres VARCHAR(100),
	provincie VARCHAR(45),

	PRIMARY KEY(filiaalnr)
);

CREATE TABLE Klant (
	klantnr INT,
	naam VARCHAR(45),
	adres VARCHAR(100),
	woonplaats VARCHAR(45),
	geslacht CHAR(1),
	geboortedatum DATE,

	PRIMARY KEY(fabrikantnr)
);

CREATE TABLE Monteur (
	monteurnr INT,
	naam VARCHAR(45),
	woonplaats VARCHAR(100),
	uurloon FLOAT,
	filiaal INT,

	PRIMARY KEY(fabrikantnr),
	FOREIGN KEY(filiaal) REFERENCES Filiaal(filiaalnr)
);

CREATE TABLE Fiets (
	fietsnr INT,
	soort VARCHAR(100),
	merk VARCHAR(45),
	type VARCHAR(45),
	standaardprijs FLOAT,
	inkoopprijs FLOAT,
	kleur VARCHAR(45)
	fabrikant INT,

	PRIMARY KEY(fabrikantnr)
	FOREIGN KEY(fabrikant) REFERENCES Fabrikant(fabrikantnr)
);

CREATE TABLE Fiets_Verkoop (
	fiets_verkoopnr INT,
	datum DATE,
	aantal INT,
	verkoopprijs FLOAT,
	klant INT,
	fiets INT,
	monteur INT,

	PRIMARY KEY(fabrikantnr),
	FOREIGN KEY(klant) REFERENCES Klant(klantnr),
	FOREIGN KEY(fiets) REFERENCES Fiets(fietsnr),
	FOREIGN KEY(monteur) REFERENCES Monteur(monteurnr)
);
