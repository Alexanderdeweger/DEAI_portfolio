--BikeToDrive_1_Accesoireverkoop;

CREATE TABLE Accesoireverkoop_Leverancier (
	leveranciernr INT,
	naam VARCHAR(45),
	adres VARCHAR(100),
	woonplaats VARCHAR(45),

	PRIMARY KEY(leveranciernr)
);

CREATE TABLE Accesoireverkoop_Filiaal (
	filiaalnr INT,
	naam VARCHAR(45),
	adres VARCHAR(100),
	provincie VARCHAR(45),

	PRIMARY KEY(filiaalnr)
);

CREATE TABLE Accesoireverkoop_Klant (
	klantnr INT,
	naam VARCHAR(45),
	adres VARCHAR(100),
	woonplaats VARCHAR(45),
    geslacht CHAR(1),
    geboortedatum DATE,


	PRIMARY KEY(klantnr)
);

CREATE TABLE Accesoireverkoop_Monteur (
	monteurnr INT,
	naam VARCHAR(45),
	woonplaats VARCHAR(45),
	uurloon FLOAT,
    filiaal INT,

	PRIMARY KEY(monteurnr),
    FOREIGN KEY(filiaal) REFERENCES Accesoireverkoop_Filiaal(filiaalnr)
);

CREATE TABLE Accesoireverkoop_Accessoire (
	accessoirenr INT,
	naam VARCHAR(45),
	standaardprijs FLOAT,
	inkoopprijs FLOAT,
    soort VARCHAR(45),
    leverancier INT,


	PRIMARY KEY(accessoirenr),
    FOREIGN KEY(leverancier) REFERENCES Accesoireverkoop_Leverancier(leveranciernr)
);

CREATE TABLE Accessoire_Verkoop (
	accessoire_verkoopnr INT,
	datum DATE,
	aantal INT,
	verkoopprijs FLOAT,
    klant INT,
    accessoire INT,
    monteur INT,

	PRIMARY KEY(accessoire_verkoopnr),
    FOREIGN KEY(klant) REFERENCES Accesoireverkoop_Klant(klantnr),
    FOREIGN KEY(accessoire) REFERENCES Accesoireverkoop_Accessoire(accessoirenr),
    FOREIGN KEY(monteur) REFERENCES Accesoireverkoop_Monteur(monteurnr)
);
